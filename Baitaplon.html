Đoạn mã gốc

function convertGrade(score) { 
  if (score >= 8.5) return 4.0; 
  else if (score >= 7.0) return 3.0; 
  else if (score >= 5.5) return 2.0; 
  else if (score >= 4.0) return 1.0; 
  else return 0; 
}

Code đã sửa lỗi

// Hàm chuyển đổi điểm hệ 10 sang hệ 4
function convertGrade(score) {

    // Kiểm tra dữ liệu đầu vào
    if(score < 0 || score > 10){
        return "Điểm không hợp lệ";
    }

    // Quy đổi điểm
    if(score >= 8.5){
        return 4.0;
    }
    else if(score >= 7.0){
        return 3.0;
    }
    else if(score >= 5.5){
        return 2.0;
    }
    else if(score >= 4.0){
        return 1.0;
    }
    else{
        return 0;
    }
}
Giải thích lỗi:
  • Thiếu kiểm tra dữ liệu đầu vào.
  • Điểm có thể âm hoặc lớn hơn 10.
  • Code được format lại rõ ràng hơn.
  • Thêm chú thích giúp dễ bảo trì.
